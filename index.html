var defaultTypes = [['Moneyline'], ['Spread'], ['Total'], ['Prop'], ['Future'], ['Teaser']]; settingsSheet.getRange(2, 1, defaultBooks.length, 1).setValues(defaultBooks); settingsSheet.getRange(2, 2, defaultSports.length, 1).setValues(defaultSports); settingsSheet.getRange(2, 3, defaultTypes.length, 1).setValues(defaultTypes); } var numRows = Math.max(settingsSheet.getLastRow() - 1, 1); var books = settingsSheet.getRange(2, 1, numRows, 1).getValues().flat().filter(String); var sportsList = settingsSheet.getRange(2, 2, numRows, 1).getValues().flat().filter(String); var typeList = settingsSheet.getRange(2, 3, numRows, 1).getValues().flat().filter(String); var sportRule = SpreadsheetApp.newDataValidation().requireValueInList(sportsList, true).setAllowInvalid(false).build(); var typeRule = SpreadsheetApp.newDataValidation().requireValueInList(typeList, true).setAllowInvalid(false).build(); var resultRule = SpreadsheetApp.newDataValidation().requireValueInList(['W', 'L', 'P'], true).setAllowInvalid(false).build(); var parlayRule = SpreadsheetApp.newDataValidation().requireCheckbox().build(); // --- STEP 2: SPORTSBOOK TABS --- books.forEach(function(book) { var sheet = ss.getSheetByName(book); if (!sheet) { sheet = ss.insertSheet(book); } var bMax = sheet.getMaxRows(); sheet.getRange(1, 1, 1, headers.length).setValues([headers]).setFontWeight("bold").setBackground("#f3f3f3"); // Protect existing bets from being overwritten, just ensure formatting and formulas are correct sheet.getRange("J2:J" + bMax).setFormula('=IF(H2="", "", IF(H2>0, (H2/100)*I2, (100/ABS(H2))*I2))'); sheet.getRange("L2:L" + bMax).setFormula('=IF(K2="", "", IF(K2="W", J2, IF(K2="L", -I2, 0)))'); sheet.getRange("B2:B" + bMax).setDataValidation(sportRule); sheet.getRange("C2:C" + bMax).setDataValidation(parlayRule); sheet.getRange("D2:D" + bMax).setDataValidation(typeRule); sheet.getRange("K2:K" + bMax).setDataValidation(resultRule); sheet.getRange("I2:J" + bMax).setNumberFormat("$#,##0.00"); sheet.getRange("L2:L" + bMax).setNumberFormat("$#,##0.00"); var bookRules = []; bookRules.push(SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('W').setBackground('#d9ead3').setFontColor('#274e13').setRanges([sheet.getRange("K2:K" + bMax)]).build()); bookRules.push(SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo('L').setBackground('#f4cccc').setFontColor('#cc0000').setRanges([sheet.getRange("K2:K" + bMax)]).build()); sportsList.forEach(function(s, i) { bookRules.push(SpreadsheetApp.newConditionalFormatRule().whenTextEqualTo(s).setBackground(getSportColor(i)).setRanges([sheet.getRange("B2:B" + bMax)]).build()); }); sheet.setConditionalFormatRules(bookRules); }); // --- STEP 3: ALL_BETS MASTER --- var allBetsSheet = ss.getSheetByName('All_Bets') || ss.insertSheet('All_Bets'); allBetsSheet.clear().getRange(1, 1, 1, masterHeaders.length).setValues([masterHeaders]).setFontWeight("bold").setBackground("#f3f3f3"); // The master formula that pulls from all book tabs dynamically var vstackArrays = books.map(function(b) { return "{ARRAYFORMULA(IF('" + b + "'!A2:A<>\"\", \"" + b + "\", \"\")), '" + b + "'!A2:L}"; }); allBetsSheet.getRange("A2").setFormula('=IFERROR(QUERY(VSTACK(' + vstackArrays.join('; ') + '), "Select * where Col2 is not null order by Col2 desc"), "No bets logged.")'); var aMax = Math.max(allBetsSheet.getMaxRows(), 100); allBetsSheet.getRange("B2:B" + aMax).setNumberFormat("mm/dd/yyyy"); allBetsSheet.getRange("J2:K" + aMax).setNumberFormat("$#,##0.00"); allBetsSheet.getRange("M2:M" + aMax).setNumberFormat("$#,##0.00"); // --- STEP 4: MAIN DASHBOARD --- var dashSheet = ss.getSheetByName('Main_Dashboard') || ss.insertSheet('Main_Dashboard'); dashSheet.clear().getRange("A1").setValue("Overall Performance").setFontWeight("bold").setFontSize(14); dashSheet.getRange("A2:B5").setValues([["Total Wagered:", "=SUM(All_Bets!J:J)"], ["Total P/L:", "=SUM(All_Bets!M:M)"], ["Total ROI:", "=IFERROR(SUM(All_Bets!M:M)/SUM(All_Bets!J:J), 0)"], ["Record:", '=COUNTIFS(All_Bets!L:L,"W")&"-"&COUNTIFS(All_Bets!L:L,"L")&"-"&COUNTIFS(All_Bets!L:L,"P")']]); dashSheet.getRange("B2:B3").setNumberFormat("$#,##0.00"); dashSheet.getRange("B4").setNumberFormat("0.00%"); dashSheet.getRange("A7").setValue("Bet Style Performance").setFontWeight("bold").setFontSize(14); dashSheet.getRange("A8:B11").setValues([ ["Straight P/L:", "=SUMIFS(All_Bets!M:M, All_Bets!D:D, FALSE)"], ["Straight Record:", '=COUNTIFS(All_Bets!D:D, FALSE, All_Bets!L:L, "W") & "-" & COUNTIFS(All_Bets!D:D, FALSE, All_Bets!L:L, "L") & "-" & COUNTIFS(All_Bets!D:D, FALSE, All_Bets!L:L, "P")'], ["Parlay P/L:", "=SUMIFS(All_Bets!M:M, All_Bets!D:D, TRUE)"], ["Parlay Record:", '=COUNTIFS(All_Bets!D:D, TRUE, All_Bets!L:L, "W") & "-" & COUNTIFS(All_Bets!D:D, TRUE, All_Bets!L:L, "L") & "-" & COUNTIFS(All_Bets!D:D, TRUE, All_Bets!L:L, "P")'] ]); dashSheet.getRange("B8").setNumberFormat("$#,##0.00"); dashSheet.getRange("B10").setNumberFormat("$#,##0.00"); dashSheet.getRange("D1").setValue("Performance by Book").setFontWeight("bold").setFontSize(14); dashSheet.getRange("D2:F2").setValues([["Book", "P/L", "ROI"]]).setBackground("#f3f3f3"); var bookRows = books.map(function(b) { return [b, '=SUMIFS(All_Bets!M:M, All_Bets!A:A, "'+b+'")', '=IFERROR(SUMIFS(All_Bets!M:M, All_Bets!A:A, "'+b+'")/SUMIFS(All_Bets!J:J, All_Bets!A:A, "'+b+'"),0)']; }); dashSheet.getRange(3, 4, bookRows.length, 3).setValues(bookRows); dashSheet.getRange(3, 5, bookRows.length, 1).setNumberFormat("$#,##0.00"); dashSheet.getRange(3, 6, bookRows.length, 1).setNumberFormat("0.00%"); dashSheet.getRange("H1").setValue("Performance by Sport").setFontWeight("bold").setFontSize(14); dashSheet.getRange("H2:L2").setValues([["Sport", "Wagered", "P/L", "ROI", "Win %"]]).setBackground("#f3f3f3"); var sportRows = sportsList.map(function(s) { return [s, '=SUMIFS(All_Bets!J:J, All_Bets!C:C, "'+s+'")', '=SUMIFS(All_Bets!M:M, All_Bets!C:C, "'+s+'")', '=IFERROR(SUMIFS(All_Bets!M:M, All_Bets!C:C, "'+s+'")/SUMIFS(All_Bets!J:J, All_Bets!C:C, "'+s+'"),0)', '=IFERROR(COUNTIFS(All_Bets!C:C,"'+s+'",All_Bets!L:L,"W")/(COUNTIFS(All_Bets!C:C,"'+s+'",All_Bets!L:L,"W")+COUNTIFS(All_Bets!C:C,"'+s+'",All_Bets!L:L,"L")),0)']; }); dashSheet.getRange(3, 8, sportRows.length, 5).setValues(sportRows); dashSheet.getRange(3, 9, sportRows.length, 2).setNumberFormat("$#,##0.00"); dashSheet.getRange(3, 11, sportRows.length, 2).setNumberFormat("0.00%"); dashSheet.getRange("A15").setValue("P/L by Bet Type (Per Sport)").setFontWeight("bold").setFontSize(14); dashSheet.getRange("A16").setFormula('=IFERROR(QUERY(All_Bets!A:M, "Select C, SUM(M) where C is not null and C != \'Sport\' group by C pivot E", 1), "Awaiting Data...")'); dashSheet.getRange("A16:F16").setFontWeight("bold").setBackground("#f3f3f3"); dashSheet.getRange("H15").setValue("P/L by Bet Type (Per Book)").setFontWeight("bold").setFontSize(14); dashSheet.getRange("H16").setFormula('=IFERROR(QUERY(All_Bets!A:M, "Select A, SUM(M) where A is not null and A != \'Book\' group by A pivot E", 1), "Awaiting Data...")'); dashSheet.getRange("H16:N16").setFontWeight("bold").setBackground("#f3f3f3"); dashSheet.getRange("A23").setValue("🔍 Player / Team Lookup:").setFontWeight("bold").setFontSize(14); dashSheet.getRange("B23:D23").merge(); dashSheet.getRange("B23").setBackground("#fff2cc"); dashSheet.getRange("A24:C24").setValues([["Wagered", "Net P/L", "Record (W-L-P)"]]).setFontWeight("bold").setBackground("#f3f3f3"); dashSheet.getRange("A25").setFormula('=IF(B23="", "", SUMIFS(All_Bets!J:J, All_Bets!F:F, "*"&B23&"*") + SUMIFS(All_Bets!J:J, All_Bets!G:G, "*"&B23&"*"))'); dashSheet.getRange("B25").setFormula('=IF(B23="", "", SUMIFS(All_Bets!M:M, All_Bets!F:F, "*"&B23&"*") + SUMIFS(All_Bets!M:M, All_Bets!G:G, "*"&B23&"*"))'); dashSheet.getRange("C25").setFormula('=IF(B23="", "", (COUNTIFS(All_Bets!F:F, "*"&B23&"*", All_Bets!L:L, "W") + COUNTIFS(All_Bets!G:G, "*"&B23&"*", All_Bets!L:L, "W")) & "-" & (COUNTIFS(All_Bets!F:F, "*"&B23&"*", All_Bets!L:L, "L") + COUNTIFS(All_Bets!G:G, "*"&B23&"*", All_Bets!L:L, "L")) & "-" & (COUNTIFS(All_Bets!F:F, "*"&B23&"*", All_Bets!L:L, "P") + COUNTIFS(All_Bets!G:G, "*"&B23&"*", All_Bets!L:L, "P")))'); dashSheet.getRange("A25:B25").setNumberFormat("$#,##0.00"); dashSheet.getRange("B17:F50").setNumberFormat("$#,##0.00"); dashSheet.getRange("I17:N50").setNumberFormat("$#,##0.00"); // --- STEP 5: DAILY P/L --- var dailySheet = ss.getSheetByName('Daily_PL') || ss.insertSheet('Daily_PL'); dailySheet.clear().getRange("A1").setValue("Daily Performance").setFontWeight("bold").setFontSize(14); dailySheet.getRange("A2").setFormula('=QUERY(All_Bets!A:M, "Select B, SUM(M) where B is not null group by B label B \'Date\', SUM(M) \'Daily P/L\'", 1)'); dailySheet.getRange("B:B").setNumberFormat("$#,##0.00"); // --- STEP 6: LEADERBOARD --- var lbSheet = ss.getSheetByName('Leaderboard') || ss.insertSheet('Leaderboard'); lbSheet.clear().getRange("A1").setValue("Most Profitable Entities").setFontWeight("bold").setFontSize(14); lbSheet.getRange("A2").setFormula('=IFERROR(QUERY(VSTACK({All_Bets!F2:F, All_Bets!M2:M}; {All_Bets!G2:G, All_Bets!M2:M}), "Select Col1, COUNT(Col1), SUM(Col2) where Col1 is not null and Col1 != \'Player/Team 1\' and Col1 != \'Team 2 (Opt)\' group by Col1 order by SUM(Col2) desc label Col1 \'Entity\', COUNT(Col1) \'Bets\', SUM(Col2) \'P/L\'", 0), "Awaiting data")'); lbSheet.getRange("E1").setValue("Least Profitable Entities").setFontWeight("bold").setFontSize(14); lbSheet.getRange("E2").setFormula('=IFERROR(QUERY(VSTACK({All_Bets!F2:F, All_Bets!M2:M}; {All_Bets!G2:G, All_Bets!M2:M}), "Select Col1, COUNT(Col1), SUM(Col2) where Col1 is not null and Col1 != \'Player/Team 1\' and Col1 != \'Team 2 (Opt)\' group by Col1 order by SUM(Col2) asc label Col1 \'Entity\', COUNT(Col1) \'Bets\', SUM(Col2) \'P/L\'", 0), "Awaiting data")'); lbSheet.getRange("C:C").setNumberFormat("$#,##0.00"); lbSheet.getRange("G:G").setNumberFormat("$#,##0.00"); // --- STEP 7: REORDER TABS --- var order = ['Main_Dashboard', 'All_Bets', 'Daily_PL', 'Leaderboard']; books.forEach(function(b) { order.push(b); }); order.push('Settings'); order.forEach(function(name, i) { var s = ss.getSheetByName(name); if(s) { ss.setActiveSheet(s); ss.moveActiveSheet(i+1); }}); ss.setActiveSheet(ss.getSheetByName('Main_Dashboard')); ss.toast("Done!", "Success", 5); } catch(e) { ui.alert("Error: " + e.message); } } // --- UPDATED POST: INSERTS AT TOP (ROW 2) --- function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); var sheet = ss.getSheetByName(data.book); if (!sheet) return ContentService.createTextOutput(JSON.stringify({"status": "error"})).setMimeType(ContentService.MimeType.JSON); // Insert new row at index 2 (pushes others down) sheet.insertRowBefore(2); var newRow = [data.date, data.sport, data.parlay, data.betType, data.team1, data.team2, data.details, data.odds, data.wager, "", data.result, ""]; sheet.getRange(2, 1, 1, 12).setValues([newRow]); // Set formulas for the new row sheet.getRange("J2").setFormula('=IF(H2="", "", IF(H2>0, (H2/100)*I2, (100/ABS(H2))*I2))'); sheet.getRange("L2").setFormula('=IF(K2="", "", IF(K2="W", J2, IF(K2="L", -I2, 0)))'); return ContentService.createTextOutput(JSON.stringify({"status": "success"})).setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({"status": "error", "message": error.message})).setMimeType(ContentService.MimeType.JSON); } } // --- SENDS STATS BACK TO THE MOBILE APP --- function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var dash = ss.getSheetByName('Main_Dashboard'); var stats = { status: "success", wagered: dash.getRange("B2").getDisplayValue(), pl: dash.getRange("B3").getDisplayValue(), roi: dash.getRange("B4").getDisplayValue(), record: dash.getRange("B5").getDisplayValue() }; return ContentService.createTextOutput(JSON.stringify(stats)).setMimeType(ContentService.MimeType.JSON); } catch (e) { return ContentService.createTextOutput(JSON.stringify({"status": "error"})).setMimeType(ContentService.MimeType.JSON); } }
