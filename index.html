            if (type === 'ENG') { bets = rawBetsData.filter(b => b.details && b.details.toUpperCase().includes('ENG')); title = "ENG Tracker History"; } 

            else if (type === 'GOAL') { bets = rawBetsData.filter(b => b.details && goalKws.some(kw => b.details.toUpperCase().includes(kw))); title = "Goal Props History"; }

            bets = bets.sort((a, b) => new Date(b.date) - new Date(a.date)); openModal(title, bets, false);

        }


        function closeModal() { document.getElementById('betsModal').style.display = 'none'; }

        window.onclick = function(e) { if (e.target == document.getElementById('betsModal')) closeModal(); }


        function buildPivotTable(dataArray) {

            if (!dataArray || dataArray.length < 2) return '<p style="font-size:13px;">Awaiting Data...</p>';

            let html = '<div class="table-wrapper"><table><tr>'; let colCount = 0; dataArray[0].forEach(h => { if (h !== "") { html += `<th>${h}</th>`; colCount++; } }); html += '</tr>';

            for (let i = 1; i < dataArray.length; i++) {

                html += '<tr>'; for (let j = 0; j < colCount; j++) { let cell = dataArray[i][j] || "", style = (j > 0 && cell !== "") ? ` class="text-right" style="color:${getColor(cell)}; font-weight:bold;"` : (j > 0 ? ' class="text-right"' : ''); html += `<td${style}>${cell}</td>`; } html += '</tr>';

            }

            html += '</table></div>'; return html;

        }


        function renderLeaderboard() {

            const filter = document.getElementById('lbSportFilter').value; let entities = {};

            rawBetsData.forEach(bet => {

                if (filter !== 'ALL' && bet.sport !== filter) return;

                ['team1', 'team2'].forEach(t => { let team = bet[t]; if (team && team.trim() !== "") { if (!entities[team]) entities[team] = { bets: 0, pl: 0 }; entities[team].bets += 1; entities[team].pl += bet.pl; } });

            });

            let sorted = Object.keys(entities).map(k => [k, entities[k].bets, entities[k].pl]).sort((a, b) => b[2] - a[2]);

            if (sorted.length === 0) { document.getElementById('tableMostProf').innerHTML = '<p style="font-size:13px;">No bets logged.</p>'; return; }

            let html = '<div class="table-wrapper"><table><tr><th>Player/Team</th><th class="text-right">Bets</th><th class="text-right">P/L</th></tr>';

            for (let i = 0; i < sorted.length; i++) { let entity = sorted[i][0], bets = sorted[i][1], plText = formatPL(sorted[i][2]); html += `<tr class="clickable-row" onclick="showEntityBets('${entity.replace(/'/g, "\\'")}')"><td style="text-decoration: underline; text-decoration-style: dotted; text-underline-offset: 3px;">${entity}</td><td class="text-right">${bets}</td><td class="text-right" style="color:${getColor(plText)}; font-weight:bold;">${plText}</td></tr>`; }

            document.getElementById('tableMostProf').innerHTML = html + '</table></div>';

        }


        function renderHallOfFame() {

            let bestDay = null, bestDayVal = -Infinity, weeklyTotals = {};

            for (const [dateStr, pl] of Object.entries(dailyPLData)) {

                if (pl > bestDayVal) { bestDayVal = pl; bestDay = dateStr; }

                let d = new Date(dateStr + "T00:00:00"), day = d.getDay(), sun = new Date(d); sun.setDate(d.getDate() - day);

                let weekKey = `${sun.getFullYear()}-${String(sun.getMonth()+1).padStart(2,'0')}-${String(sun.getDate()).padStart(2,'0')}`; weeklyTotals[weekKey] = (weeklyTotals[weekKey] || 0) + pl;

            }

            let bestWeek = null, bestWeekVal = -Infinity;

            for (const [w, pl] of Object.entries(weeklyTotals)) { if (pl > bestWeekVal) { bestWeekVal = pl; bestWeek = w; } }

            document.getElementById('hofBestDay').textContent = bestDayVal > 0 ? `${bestDay} (${formatPL(bestDayVal)})` : '--'; document.getElementById('hofBestDay').style.color = bestDayVal > 0 ? getColor(formatPL(bestDayVal)) : ''; document.getElementById('hofBestWeek').textContent = bestWeekVal > 0 ? `Week of ${bestWeek} (${formatPL(bestWeekVal)})` : '--'; document.getElementById('hofBestWeek').style.color = bestWeekVal > 0 ? getColor(formatPL(bestWeekVal)) : '';

            let monsters = rawBetsData.filter(b => b.pl >= 300).sort((a,b) => b.pl - a.pl);

            if (monsters.length === 0) { document.getElementById('hofMonsterBets').innerHTML = '<p style="font-size:13px;">No monster hits yet.</p>'; } 

            else { let html = '<div class="table-wrapper"><table><tr><th>Date</th><th>Player/Team</th><th>Details</th><th class="text-right">Wager</th><th class="text-right">Won</th></tr>'; monsters.forEach(m => { html += `<tr><td>${m.date.substring(5)}</td><td>${m.team1}</td><td>${m.details}</td><td class="text-right">$${m.wager.toFixed(2)}</td><td class="text-right" style="color:var(--win-text); font-weight:bold;">${formatPL(m.pl)}</td></tr>`; }); document.getElementById('hofMonsterBets').innerHTML = html + '</table></div>'; }

        }


        function updateCalendarSummaries(month, year) {

            let weekPl = 0, monthPl = 0, ytdPl = 0; const today = new Date(), dayOfWeek = today.getDay();

            for(let i=0; i<7; i++) { let d = new Date(today); d.setDate(today.getDate() - dayOfWeek + i); let dStr = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`; if(dailyPLData[dStr]) weekPl += dailyPLData[dStr]; }

            const viewMonthPrefix = `${year}-${String(month+1).padStart(2,'0')}`, viewYearPrefix = `${year}-`;

            for (const [dateStr, pl] of Object.entries(dailyPLData)) { if (dateStr.startsWith(viewYearPrefix)) { ytdPl += pl; if (dateStr.startsWith(viewMonthPrefix)) monthPl += pl; } }

            document.getElementById('calWeekPL').textContent = formatPL(weekPl); document.getElementById('calWeekPL').style.color = getColor(formatPL(weekPl)); document.getElementById('calMonthPL').textContent = formatPL(monthPl); document.getElementById('calMonthPL').style.color = getColor(formatPL(monthPl)); document.getElementById('calYtdPL').textContent = formatPL(ytdPl); document.getElementById('calYtdPL').style.color = getColor(formatPL(ytdPl));

        }

        function changeMonth(dir) { currentMonth += dir; if(currentMonth > 11) { currentMonth = 0; currentYear++; } if(currentMonth < 0) { currentMonth = 11; currentYear--; } renderCalendar(currentMonth, currentYear); }


        function renderCalendar(month, year) {

            const grid = document.getElementById('calendarGrid'); const monthNames = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"];

            document.getElementById('monthLabel').textContent = `${monthNames[month]} ${year}`; grid.innerHTML = '';

            const firstDay = new Date(year, month, 1).getDay(); const daysInMonth = new Date(year, month + 1, 0).getDate();

            ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'].forEach(d => { grid.innerHTML += `<div class="day-name">${d}</div>`; });

            for (let i = 0; i < firstDay; i++) { grid.innerHTML += `<div class="day-cell empty"></div>`; }

            for (let i = 1; i <= daysInMonth; i++) {

                const dStr = `${year}-${String(month+1).padStart(2, '0')}-${String(i).padStart(2, '0')}`; let pl = dailyPLData[dStr] || 0;

                let plText = pl === 0 ? '-' : (pl > 0 ? `+$${Math.round(pl)}` : `-$${Math.round(Math.abs(pl))}`); let cls = pl > 0 ? 'profit' : (pl < 0 ? 'loss' : '');

                grid.innerHTML += `<div class="day-cell ${cls}" onclick="showDayBets('${dStr}')"><div class="date-num">${i}</div><div class="pl-amt">${plText}</div></div>`;

            }

            updateCalendarSummaries(month, year); renderHallOfFame(); 

            setChartRange('MONTH');

        }


        function submitBet() {

            const btn = document.getElementById('submitBtn'), statusDiv = document.getElementById('status'), keepSlip = document.getElementById('keepSlip').checked;

            const payload = { book: document.getElementById('book').value, date: document.getElementById('date').value, sport: document.getElementById('sport').value, parlay: document.getElementById('parlay').checked, betType: document.getElementById('betType').value, team1: document.getElementById('team1').value, team2: document.getElementById('team2').value, details: document.getElementById('details').value, odds: parseFloat(document.getElementById('odds').value), wager: parseFloat(document.getElementById('wager').value), result: document.getElementById('result').value };

            if (!payload.team1 || isNaN(payload.odds) || isNaN(payload.wager)) { statusDiv.textContent = "Fill Team 1, Odds, and Wager."; statusDiv.className = "error"; statusDiv.style.display = "block"; return; }

            let finalData = editingBet ? { action: "edit", old: editingBet, new: payload } : { action: "log", ...payload };


            btn.disabled = true; btn.textContent = "SENDING...";

            fetch(WEB_APP_URL, { method: 'POST', body: JSON.stringify(finalData) }).then(res => res.json()).then(data => {

                if (data.status === "success") {

                    statusDiv.textContent = editingBet ? "Bet Updated!" : "Bet Logged!"; statusDiv.className = "success";

                    if (editingBet || !keepSlip) { cancelEdit(); }

                    dataFetched = false; fetchMasterData(true);

                } else { statusDiv.textContent = "Error: " + data.message; statusDiv.className = "error"; btn.textContent = editingBet ? "UPDATE BET" : "LOG BET"; }

                statusDiv.style.display = "block"; btn.disabled = false; 

            });

        }
